```go
// Walk traverses the VFS tree rooted at `root` depth-first, calling walkFn for every
// node. `dir` is non-nil for a directory node; `file` is non-nil for a file node.
```

RFC 4918 §9.8.8 — 207 Multi-Status body shape for partial COPY failures:
```xml
<?xml version="1.0" encoding="utf-8"?>
<D:multistatus xmlns:D="DAV:">
  <D:response>
    <D:href>/dav/files/dest/subdir/failedfile.txt</D:href>
    <D:status>HTTP/1.1 507 Insufficient Storage</D:status>
  </D:response>
</D:multistatus>
```
Only failed entries are listed. The root destination collection is NOT listed if created successfully.
Already-copied children stay in place (no rollback).

Phase 1 XML convention (from 01-02 decisions):
Root element must be manually written as `<D:multistatus xmlns:D="DAV:">` to avoid encoding/xml
emitting redundant `xmlns="DAV:"` on every child. See web/webdav/propfind.go for the established pattern.

    Step C — Implement `handleCopyDir`. Pseudocode:
    ```
    func handleCopyDir(c, fs, inst, srcDir, dstPath, depth, overwrite) error {
        // 1. Check destination existence.
        destExisted := false
        _, existingDstFile, err := fs.DirOrFileByPath(dstPath)
        existingDstDir, _, _ := fs.DirOrFileByPath(dstPath)  // or reuse the first result
        if err != nil && !errors.Is(err, os.ErrNotExist) {
            return mapVFSWriteError(c, err)
        }
        if existingDstDir != nil || existingDstFile != nil {
            if !overwrite {
                return sendWebDAVError(c, http.StatusPreconditionFailed, "precondition-failed")
            }
            destExisted = true
            if existingDstFile != nil {
                _, err = vfs.TrashFile(fs, existingDstFile)
            } else {
                _, err = vfs.TrashDir(fs, existingDstDir)
            }
            if err != nil { return mapVFSWriteError(c, err) }
        }

        // 2. Resolve destination parent.
        dstParent, err := fs.DirByPath(path.Dir(dstPath))
        if err != nil {
            if errors.Is(err, os.ErrNotExist) {
                return sendWebDAVError(c, http.StatusConflict, "conflict")
            }
            return mapVFSWriteError(c, err)
        }

        // 3. Create destination root directory (single level, not MkdirAll — race).
        _, err = vfs.Mkdir(fs, dstPath, nil)
        if err != nil {
            return mapVFSWriteError(c, err)
        }

        // 4. Depth:0 is done — return.
        if depth == "0" {
            if destExisted { return c.NoContent(http.StatusNoContent) }
            return c.NoContent(http.StatusCreated)
        }

        // 5. Recursive walk.
        type copyError struct { hrefPath string; status int }
        var failures []copyError
        srcPathStr := srcDir.Fullpath  // or reconstruct via path.Clean
        walkFn := func(name string, dir *vfs.DirDoc, file *vfs.FileDoc, werr error) error {
            if werr != nil {
                failures = append(failures, copyError{hrefPath: name, status: http.StatusInternalServerError})
                return nil // do NOT abort — pitfall C
            }
            // Skip the root itself (already created above).
            if name == srcPathStr {
                return nil
            }
            // Compute destination path.
            rel := strings.TrimPrefix(name, srcPathStr)
            destChild := path.Join(dstPath, rel)

            if dir != nil {
                _, mkErr := vfs.Mkdir(fs, destChild, nil)
                if mkErr != nil && !errors.Is(mkErr, os.ErrExist) {
                    failures = append(failures, copyError{hrefPath: destChild, status: 500})
                }
                return nil
            }

            // file != nil — copy it.
            parentDir, derr := fs.DirByPath(path.Dir(destChild))
            if derr != nil {
                failures = append(failures, copyError{hrefPath: destChild, status: 409})
                return nil
            }
            newdoc := vfs.CreateFileDocCopy(file, parentDir.ID(), path.Base(destChild))
            var copyErr error
            if file.Mime == consts.NoteMimeType {
                copyErr = note.CopyFile(inst, file, newdoc)
            } else {
                copyErr = fs.CopyFile(file, newdoc)
            }
            if copyErr != nil {
                failures = append(failures, copyError{hrefPath: destChild, status: httpStatusForVFSErr(copyErr)})
            }
            return nil
        }

        if err := vfs.Walk(fs, srcPathStr, walkFn); err != nil {
            // ErrWalkOverflow or catastrophic failure -> 500
            return mapVFSWriteError(c, err)
        }

        // 6. Response.
        if len(failures) > 0 {
            return sendCopyMultiStatus(c, failures)
        }
        if destExisted { return c.NoContent(http.StatusNoContent) }
        return c.NoContent(http.StatusCreated)
    }
    ```

    Step D — Implement `sendCopyMultiStatus(c echo.Context, failures []copyError) error`. Body builder:
    ```go
    var buf bytes.Buffer
    buf.WriteString(`<?xml version="1.0" encoding="utf-8"?>` + "\n")
    buf.WriteString(`<D:multistatus xmlns:D="DAV:">` + "\n")
    for _, f := range failures {
        buf.WriteString("  <D:response>\n")
        buf.WriteString("    <D:href>")
        xml.EscapeText(&buf, []byte(f.hrefPath))
        buf.WriteString("</D:href>\n")
        buf.WriteString(fmt.Sprintf("    <D:status>HTTP/1.1 %d %s</D:status>\n", f.status, http.StatusText(f.status)))
        buf.WriteString("  </D:response>\n")
    }
    buf.WriteString(`</D:multistatus>` + "\n")
    h := c.Response().Header()
    h.Set(echo.HeaderContentType, "application/xml; charset=utf-8")
    h.Set(echo.HeaderContentLength, strconv.Itoa(buf.Len()))
    return c.Blob(http.StatusMultiStatus, "application/xml; charset=utf-8", buf.Bytes())
    ```
