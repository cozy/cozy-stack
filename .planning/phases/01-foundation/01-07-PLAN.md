    Step 1: Create web/webdav/propfind.go with handlePropfind and its helpers.

    Key structure:
    ```go
    package webdav

    import (
        "errors"
        "io"
        "net/http"
        "os"
        "path"

        "github.com/cozy/cozy-stack/model/permission"
        "github.com/cozy/cozy-stack/model/vfs"
        "github.com/cozy/cozy-stack/web/middlewares"
        "github.com/labstack/echo/v4"
    )

    const propfindDirIteratorBatch = 200 // CONTEXT.md: overrides Cozy default 100, capped by iter.go at 256

    func handlePropfind(c echo.Context) error {
        // 1. Depth header first — before any VFS call
        depth := c.Request().Header.Get("Depth")
        if depth == "" {
            depth = "1" // RFC 4918 default for PROPFIND when Depth omitted is infinity,
                        // but we map absent → "1" for safety. Note: some impls default to 0 for file requests;
                        // since we can't know the target type yet, "1" is the least surprising default for
                        // collection-heavy clients. Adjust if spec compliance demands infinity rejection here.
        }
        if depth == "infinity" || depth == "Infinity" {
            auditLog(c, "propfind depth infinity rejected", c.Param("*"))
            return sendWebDAVError(c, http.StatusForbidden, "propfind-finite-depth")
        }
        if depth != "0" && depth != "1" {
            return sendWebDAVError(c, http.StatusBadRequest, "bad-depth")
        }

        // 2. Path resolution (security boundary)
        rawParam := c.Param("*")
        vfsPath, err := davPathToVFSPath(rawParam)
        if err != nil {
            auditLog(c, "propfind path rejected", rawParam)
            return sendWebDAVError(c, http.StatusForbidden, "forbidden")
        }

        // 3. VFS lookup
        inst := middlewares.GetInstance(c)
        dirDoc, fileDoc, err := inst.VFS().DirOrFileByPath(vfsPath)
        if err != nil {
            if errors.Is(err, os.ErrNotExist) {
                return sendWebDAVError(c, http.StatusNotFound, "not-found")
            }
            return err
        }

        // 4. Permission check — AUTH-05
        var target permission.Fetcher
        if dirDoc != nil {
            target = dirDoc
        } else {
            target = fileDoc
        }
        if err := middlewares.AllowVFS(c, permission.GET, target); err != nil {
            auditLog(c, "propfind out-of-scope", vfsPath)
            return sendWebDAVError(c, http.StatusForbidden, "forbidden")
        }

        // 5. Body parsing (we currently ignore the body and always return allprop;
        //    this is spec-compliant for v1 per CONTEXT.md)
        _ = c.Request().Body

        // 6. Build responses. Because SEC-05 requires Content-Length for strict
        //    clients on small responses but Depth:1 streams, we branch:
        //    - Depth:0 → build in bytes.Buffer, set Content-Length
        //    - Depth:1 → stream, accept chunked
        responses := make([]Response, 0, 1)
        if fileDoc != nil {
            responses = append(responses, buildResponseForFile(c, fileDoc))
        } else {
            responses = append(responses, buildResponseForDir(c, dirDoc))
            if depth == "1" {
                iter := inst.VFS().DirIterator(dirDoc, &vfs.IteratorOptions{ByFetch: propfindDirIteratorBatch})
                for {
                    d, f, iErr := iter.Next()
                    if errors.Is(iErr, vfs.ErrIteratorDone) {
                        break
                    }
                    if iErr != nil {
                        return iErr
                    }
                    if d != nil {
                        responses = append(responses, buildResponseForDir(c, d))
                    } else if f != nil {
                        responses = append(responses, buildResponseForFile(c, f))
                    }
                }
            }
        }

        body, err := marshalMultistatus(responses)
        if err != nil {
            return err
        }
        h := c.Response().Header()
        h.Set(echo.HeaderContentType, `application/xml; charset="utf-8"`)
        h.Set(echo.HeaderContentLength, strconv.Itoa(len(body)))
        c.Response().WriteHeader(http.StatusMultiStatus)
        _, werr := c.Response().Write(body)
        return werr
    }
    ```

    Implement buildResponseForDir and buildResponseForFile to populate Prop with the 9 live properties per CONTEXT.md:
    - resourcetype: collection for dirs, empty for files
    - displayname: DocName
    - getlastmodified: buildLastModified(UpdatedAt)
    - getetag: buildETag(MD5Sum) for files; for directories, derive from a stable hash or omit (directories have no md5sum in the VFS — use a deterministic string from DocID+UpdatedAt). The RESEARCH pitfall 5 acknowledges this.
    - getcontentlength: ByteSize (files only)
    - getcontenttype: Mime (files only, fallback application/octet-stream)
    - creationdate: buildCreationDate(CreatedAt)
    - supportedlock: &struct{}{} (always empty)
    - lockdiscovery: &struct{}{} (always empty)

    Href construction: the WebDAV href MUST be the URL path as seen by the client, i.e. `/dav/files` + vfsPath-stripped name. For nested children, it's `/dav/files/<parent>/<name>`. Use `path.Join` + leading `/dav/files`.

    Collections MUST have a trailing slash in their href per RFC 4918 §5.2.
