    1. Implement `parseDestination` in write_helpers.go:
       ```go
       func parseDestination(r *http.Request) (string, error) {
           rawDest := r.Header.Get("Destination")
           if rawDest == "" {
               return "", errMissingDestination
           }
           u, err := url.Parse(rawDest)
           if err != nil {
               return "", errInvalidDestination
           }
           // u.Path is already URL-decoded by url.Parse
           const prefix = "/dav/files"
           if !strings.HasPrefix(u.Path, prefix) {
               return "", errInvalidDestination
           }
           // Strip the /dav/files prefix and pass remainder through davPathToVFSPath
           // for traversal validation and normalization
           param := strings.TrimPrefix(u.Path, prefix)
           return davPathToVFSPath(param)
       }
       ```
       Add necessary imports: `"net/http"`, `"net/url"`.

    2. Implement `handleMove` in move.go:
       a. Extract source VFS path: `srcPath, err := davPathToVFSPath(c.Param("*"))`; on error -> 403

       b. Parse destination: `dstPath, err := parseDestination(c.Request())`
          - errMissingDestination -> 400 "bad-request" (or use a more specific condition)
          - errInvalidDestination -> 502 "bad-gateway" (RFC 4918 section 9.9.4 for cross-server)
          - ErrPathTraversal -> auditLog "destination traversal" + 403 "forbidden"

       c. Check trash fence on destination: `if isInTrash(dstPath)` -> auditLog "move to trash attempt" + 403 "forbidden"

       d. Resolve source: `srcDir, srcFile, err := inst.VFS().DirOrFileByPath(srcPath)`
          - os.ErrNotExist -> 404
          - other error -> mapVFSWriteError

       e. Determine Overwrite semantics:
          ```go
          overwrite := true // RFC 4918 default = T (NOT F like x/net/webdav bug #66059)
          if ovr := c.Request().Header.Get("Overwrite"); ovr == "F" {
              overwrite = false
          }
          ```

       f. Check if destination exists: `dstDir, dstFile, err := inst.VFS().DirOrFileByPath(dstPath)`
          - If destination exists and overwrite == false -> 412 "precondition-failed"
          - If destination exists and overwrite == true -> trash the destination first:
            ```go
            if dstFile != nil {
                _, err = vfs.TrashFile(inst.VFS(), dstFile)
            } else if dstDir != nil {
                _, err = vfs.TrashDir(inst.VFS(), dstDir)
            }
            if err != nil {
                return mapVFSWriteError(c, err)
            }
            ```

       g. Resolve destination parent directory:
          ```go
          dstParent, err := inst.VFS().DirByPath(path.Dir(dstPath))
          if err != nil {
              if errors.Is(err, os.ErrNotExist) {
                  return sendWebDAVError(c, http.StatusConflict, "conflict")
              }
              return mapVFSWriteError(c, err)
          }
          ```

       h. Build DocPatch and execute move:
          ```go
          newName := path.Base(dstPath)
          newDirID := dstParent.ID()
          patch := &vfs.DocPatch{
              Name:  &newName,
              DirID: &newDirID,
          }
          if srcFile != nil {
              _, err = vfs.ModifyFileMetadata(inst.VFS(), srcFile, patch)
          } else {
              _, err = vfs.ModifyDirMetadata(inst.VFS(), srcDir, patch)
          }
          if err != nil {
              return mapVFSWriteError(c, err)
          }
          ```

       i. Return status: if destination existed (was trashed) -> 204 No Content; if new destination -> 201 Created.
